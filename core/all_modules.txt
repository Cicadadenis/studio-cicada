### Middleware (до каждого / после каждого)  [middleware_block]
# Выполняется перед/после каждого сообщения — проверка бана, логирование

до каждого:
    # Проверяем статус бана перед каждым запросом
    получить "забанен" → забанен
    если забанен == "да":
        ответ "🚫 Ваш аккаунт заблокирован."
        вернуть
    # Автоматически логируем дату входа
    сохранить "последний_визит" = текущая_дата

после каждого:
    # Счётчик активности
    получить "активность" → акт
    если не акт:
        запомни акт = 0
    запомни акт = акт + 1
    сохранить "активность" = акт
